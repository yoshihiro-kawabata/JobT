# 開発言語
- Ruby 3.0.1
- Rails 6.0.4.8
# 実行手順
    $ git@github.com:yoshihiro-kawabata/JobT.git
    $ cd JobT
    $ bundle install
    $ rails db:create && rails db:migrate
    $ rails s
